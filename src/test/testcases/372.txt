                    for i := n downto a[n] do begin
                        b[i] := a[i] + x[i];
                        if i = a[x[i]] then
                            return x;
                        else
                            continue;
                    end
                else
                    for i := 1 to n mod a[n] do begin
                        b[i] := a[i] and then x[i] or else b[i];
                        if a[x[i]] then
                            return x;
                        else
                            break;
                    end
            return foo(foo(x),a[x[n]]);
        end
